#define filaSerial 4
#define fEtapa1 6
#define fEtapa2 5
#define clear 7

//VERIFICACION DE LOS LEDS
void setup()
{
  int pines[4]={4,6,5,7};
  for(int i=0;i<4;i++)
  {
	pinMode(pines[i], OUTPUT);
    digitalWrite(pines[i], LOW);
  }
  digitalWrite(clear, HIGH);
  digitalWrite(filaSerial, HIGH);
  digitalWrite(fEtapa1, LOW);
  digitalWrite(fEtapa1, HIGH);
  digitalWrite(fEtapa1, LOW);
  digitalWrite(fEtapa2, LOW);
  digitalWrite(fEtapa2, HIGH);
  digitalWrite(fEtapa2, LOW);

  
}

void loop()
{
  digitalWrite(filaSerial, LOW);
  digitalWrite(fEtapa1, LOW);
  digitalWrite(fEtapa1, HIGH);
  digitalWrite(fEtapa1, LOW);
  digitalWrite(fEtapa2, LOW);
  digitalWrite(fEtapa2, HIGH);
  digitalWrite(fEtapa2, LOW);
  delay(1000);
}
